    /**
     * 如何使文字实现渐变出现效果

    (function startMove() {
        var timer = null;
        clearInterval(timer);
        timer = setInterval(function () {
            var speed = 5;
            if (oMove.offsetTop >= oContent.offsetHeight) {
                clearInterval(timer);
